/**
 * server-api/workers/emailWorker.js
 *
 * Run standalone:
 *   node workers/emailWorker.js
 *
 * Or from server.js:
 *   const { startEmailWorker } = require("./workers/emailWorker");
 *   startEmailWorker();
 */

const path = require("path");
require("dotenv").config({ path: path.join(__dirname, "..", ".env") });

const fs = require("fs");
const axios = require("axios");
const FormData = require("form-data");

const db = require("../db");
const { decryptJSON } = require("../utils/crypto");

const WORKER_NAME = "emailWorker";
const POLL_MS = Number(process.env.EMAIL_WORKER_POLL_MS || 2500);
const BATCH_SIZE = Math.min(
  Number(process.env.EMAIL_WORKER_BATCH_SIZE || 25),
  100,
);

function log(...args) {
  console.log(`[${WORKER_NAME}]`, ...args);
}
function errlog(...args) {
  console.error(`[${WORKER_NAME}]`, ...args);
}

function safeStr(v) {
  return (v ?? "").toString();
}

function normalizeProvider(p) {
  return safeStr(p).trim().toLowerCase();
}

// ✅ Support BOTH camelCase + snake_case secrets
function getSecret(secrets, camelKey, snakeKey) {
  if (!secrets || typeof secrets !== "object") return "";
  const a = secrets?.[camelKey];
  const b = secrets?.[snakeKey];
  return safeStr(a || b).trim();
}

/**
 * Accept:
 *  - "/uploads/xyz.png"
 *  - "http://host/uploads/xyz.png"
 * Convert to absolute file path under server-api/uploads
 */
function parseUploadsPathFromUrl(url) {
  if (!url) return null;

  let pathname = url;
  try {
    if (/^https?:\/\//i.test(url)) {
      const u = new URL(url);
      pathname = u.pathname;
    }
  } catch {
    // ignore
  }

  if (!pathname.startsWith("/uploads/")) return null;

  // worker is /server-api/workers; uploads is /server-api/uploads
  const uploadsRoot = path.join(__dirname, "..", "uploads") + path.sep;

  const rel = pathname.replace(/^\/+/, ""); // "uploads/xyz.png"
  const abs = path.join(__dirname, "..", rel); // server-api/uploads/xyz.png
  const absNorm = path.normalize(abs);

  if (!absNorm.startsWith(uploadsRoot)) return null;
  return absNorm;
}

async function getActiveEmailConnection() {
  // Prefer DB connection (encrypted secrets)
  try {
    const r = await db.query(
      `SELECT provider, secrets_enc
       FROM email_connections
       WHERE is_active = TRUE
       ORDER BY updated_at DESC
       LIMIT 1`,
    );

    if (r.rows?.length) {
      const provider = normalizeProvider(r.rows[0].provider);
      const secrets = decryptJSON(r.rows[0].secrets_enc);
      return { provider, secrets };
    }
  } catch (e) {
    errlog("getActiveEmailConnection error:", e?.message || e);
  }

  // Fallback to env (Mailgun)
  const provider = "mailgun";
  const secrets = {
    apiKey: process.env.MAILGUN_API_KEY || "",
    domain: process.env.MAILGUN_DOMAIN || "",
    fromEmail: process.env.MAILGUN_FROM_EMAIL || "",
    fromName: process.env.MAILGUN_FROM_NAME || "Bishop Robertson TV",
    region: process.env.MAILGUN_REGION || "us",
  };

  if (!secrets.apiKey || !secrets.domain || !secrets.fromEmail) return null;
  return { provider, secrets };
}

function mailgunBaseUrl(region = "us") {
  return region && region.toLowerCase() === "eu"
    ? "https://api.eu.mailgun.net"
    : "https://api.mailgun.net";
}

/**
 * ✅ Throws a clean error string that includes Mailgun's response message/body.
 * This is what will show in your "Recipients -> Error" column.
 */
function mailgunErrorToString(resOrErr) {
  const res = resOrErr?.response || resOrErr;
  const status = res?.status;
  const data = res?.data;

  if (!status) return safeStr(resOrErr?.message || "Mailgun request failed");

  if (typeof data === "string")
    return `Mailgun ${status}: ${data}`.slice(0, 1000);
  if (data?.message)
    return `Mailgun ${status}: ${safeStr(data.message)}`.slice(0, 1000);
  if (data?.error)
    return `Mailgun ${status}: ${safeStr(data.error)}`.slice(0, 1000);

  try {
    return `Mailgun ${status}: ${JSON.stringify(data)}`.slice(0, 1000);
  } catch {
    return `Mailgun ${status}: (unreadable error body)`.slice(0, 1000);
  }
}

async function sendViaMailgun({
  secrets,
  to,
  subject,
  text,
  html,
  attachments = [],
}) {
  const apiKey = getSecret(secrets, "apiKey", "api_key");
  const domain = getSecret(secrets, "domain", "domain");
  const fromName =
    getSecret(secrets, "fromName", "from_name") || "Bishop Robertson TV";
  const fromEmail = getSecret(secrets, "fromEmail", "from_email");
  const region = getSecret(secrets, "region", "region") || "us";

  if (!apiKey) throw new Error("Mailgun apiKey/api_key missing");
  if (!domain) throw new Error("Mailgun domain missing");
  if (!fromEmail) throw new Error("Mailgun fromEmail/from_email missing");

  const from = `${fromName} <${fromEmail}>`;

  const form = new FormData();
  form.append("from", from);
  form.append("to", to);
  form.append("subject", subject || "(no subject)");
  if (text) form.append("text", text);
  if (html) form.append("html", html);

  // Attachments: only local /uploads files supported
  for (const a of (Array.isArray(attachments) ? attachments : []).slice(0, 8)) {
    const filePath = parseUploadsPathFromUrl(a?.url);
    if (!filePath) continue;
    if (!fs.existsSync(filePath)) continue;

    form.append("attachment", fs.createReadStream(filePath), {
      filename: a?.name || path.basename(filePath),
      contentType: a?.mime || undefined,
    });
  }

  const base = mailgunBaseUrl(region);
  const url = `${base}/v3/${encodeURIComponent(domain)}/messages`;

  // ✅ do NOT use validateStatus to force axios to throw generic errors
  const res = await axios.post(url, form, {
    auth: { username: "api", password: apiKey },
    headers: form.getHeaders(),
    timeout: 30000,
    maxBodyLength: Infinity,
    maxContentLength: Infinity,
    validateStatus: () => true,
  });

  if (res.status < 200 || res.status >= 300) {
    const msg = mailgunErrorToString(res);
    throw new Error(msg);
  }

  return res.data;
}

async function updateJobStatus(jobId) {
  const r = await db.query(
    `SELECT
        SUM(CASE WHEN status='queued' THEN 1 ELSE 0 END) AS queued,
        SUM(CASE WHEN status='sending' THEN 1 ELSE 0 END) AS sending,
        SUM(CASE WHEN status='sent' THEN 1 ELSE 0 END) AS sent,
        SUM(CASE WHEN status='failed' THEN 1 ELSE 0 END) AS failed
     FROM email_job_items
     WHERE job_id = $1`,
    [jobId],
  );

  const s = r.rows?.[0] || {};
  const queued = Number(s.queued || 0);
  const sending = Number(s.sending || 0);
  const sent = Number(s.sent || 0);
  const failed = Number(s.failed || 0);

  let status = "queued";
  if (queued > 0 || sending > 0) status = "queued";
  else if (failed > 0) status = "failed";
  else if (sent > 0 && failed === 0) status = "sent";

  await db.query(
    `UPDATE email_jobs
     SET status = $2, updated_at = NOW()
     WHERE id = $1`,
    [jobId, status],
  );
}

async function claimQueuedItems(client) {
  const r = await client.query(
    `WITH picked AS (
        SELECT id
        FROM email_job_items
        WHERE status = 'queued'
        ORDER BY id ASC
        FOR UPDATE SKIP LOCKED
        LIMIT $1
     )
     UPDATE email_job_items it
     SET status = 'sending', updated_at = NOW()
     FROM picked
     WHERE it.id = picked.id
     RETURNING it.id, it.job_id, it.user_id, it.email`,
    [BATCH_SIZE],
  );

  return r.rows || [];
}

// ✅ Read job payload INCLUDING attachments if column exists
async function getJobPayload(jobId) {
  try {
    const r = await db.query(
      `SELECT id, subject, body_html, body_text, audience, attachments
       FROM email_jobs
       WHERE id = $1
       LIMIT 1`,
      [jobId],
    );
    return r.rows?.[0] || null;
  } catch (_e) {
    const r = await db.query(
      `SELECT id, subject, body_html, body_text, audience
       FROM email_jobs
       WHERE id = $1
       LIMIT 1`,
      [jobId],
    );
    return r.rows?.[0] || null;
  }
}

function normalizeAttachments(job) {
  const a1 = job?.attachments;
  if (Array.isArray(a1)) return a1;

  const a2 = job?.audience?.attachments;
  if (Array.isArray(a2)) return a2;

  return [];
}

async function processOnce() {
  const conn = await getActiveEmailConnection();
  if (!conn) {
    log(
      "No active email connection (DB) and MAILGUN env fallback is incomplete. Waiting…",
    );
    return;
  }

  if (normalizeProvider(conn.provider) !== "mailgun") {
    log(`Provider "${conn.provider}" not implemented in worker yet. Waiting…`);
    return;
  }

  // Helpful one-line debug (domain/from)
  const domain = getSecret(conn.secrets, "domain", "domain");
  const fromEmail = getSecret(conn.secrets, "fromEmail", "from_email");
  if (domain && fromEmail) {
    log(`Using Mailgun domain="${domain}" from="${fromEmail}"`);
  }

  const client = await db.pool.connect();
  try {
    await client.query("BEGIN");
    const items = await claimQueuedItems(client);
    await client.query("COMMIT");

    log(`Claimed ${items.length} item(s)`);

    if (!items.length) return;

    // Group claimed items by job
    const byJob = new Map();
    for (const it of items) {
      const jobId = Number(it.job_id);
      if (!byJob.has(jobId)) byJob.set(jobId, []);
      byJob.get(jobId).push(it);
    }

    for (const [jobId, jobItems] of byJob.entries()) {
      const job = await getJobPayload(jobId);

      if (!job) {
        for (const it of jobItems) {
          await db.query(
            `UPDATE email_job_items
             SET status='failed', error=$2, updated_at=NOW()
             WHERE id=$1`,
            [it.id, "Job not found"],
          );
        }
        await updateJobStatus(jobId);
        continue;
      }

      const attachments = normalizeAttachments(job);

      for (const it of jobItems) {
        const to = safeStr(it.email).trim();
        if (!to) {
          await db.query(
            `UPDATE email_job_items
             SET status='failed', error=$2, updated_at=NOW()
             WHERE id=$1`,
            [it.id, "Recipient email missing"],
          );
          continue;
        }

        try {
          await sendViaMailgun({
            secrets: conn.secrets,
            to,
            subject: safeStr(job.subject),
            text: safeStr(job.body_text),
            html: safeStr(job.body_html),
            attachments,
          });

          await db.query(
            `UPDATE email_job_items
             SET status='sent', error=NULL, sent_at=NOW(), updated_at=NOW()
             WHERE id=$1`,
            [it.id],
          );
        } catch (e) {
          const msg = safeStr(e?.message || e).slice(0, 1000);

          await db.query(
            `UPDATE email_job_items
             SET status='failed', error=$2, updated_at=NOW()
             WHERE id=$1`,
            [it.id, msg],
          );

          errlog(`Send failed to "${to}":`, msg);
        }
      }

      await updateJobStatus(jobId);
    }
  } catch (e) {
    try {
      await client.query("ROLLBACK");
    } catch {}
    errlog("processOnce error:", e?.message || e);
  } finally {
    client.release();
  }
}

function startEmailWorker() {
  log(`Starting worker (poll=${POLL_MS}ms batch=${BATCH_SIZE})`);

  if (!db?.pool?.connect) {
    errlog("DB pool missing. Check server-api/db.js exports.");
    return { stop: () => {} };
  }

  let stopped = false;
  const timer = setInterval(async () => {
    if (stopped) return;
    try {
      await processOnce();
    } catch (e) {
      errlog("loop error:", e?.message || e);
    }
  }, POLL_MS);

  // fire once immediately
  processOnce().catch((e) => errlog("initial run error:", e?.message || e));

  return {
    stop() {
      stopped = true;
      clearInterval(timer);
      log("Worker stopped.");
    },
  };
}

// If executed directly: run worker
if (require.main === module) {
  startEmailWorker();
}

module.exports = { startEmailWorker };
